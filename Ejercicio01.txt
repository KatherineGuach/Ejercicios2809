public class Ejer01 {
	
	
	public static String invertirCadena(String cadena) {

		String[] palabrasC = cadena.split(" ");
		String cadenaN = " ";
		
		for (int i=1; i < palabrasC.length - 1; i++) {
			cadenaN = cadenaN + " " + palabrasC[i];
		}
		
		cadenaN = palabrasC[palabrasC.length - 1] + " " + cadenaN + " " + palabrasC[0];
		return cadenaN;
	}
